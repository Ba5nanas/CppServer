386  void TryReceive()
387  {
388  if (_reciving)
389  return;
390 
391  if (!IsHandshaked())
392  return;
393 
394  // Async receive with the receive handler
395  _reciving = true;
396  auto self(this->shared_from_this());
397  auto async_receive_handler = make_alloc_handler(_recive_storage, [this, self](std::error_code ec, size_t size)
398  {
399  _reciving = false;
400 
401  if (!IsHandshaked())
402  return;
403 
404  // Received some data from the client
405  if (size > 0)
406  {
407  // Update statistic
408  _bytes_received += size;
409 
410  // If the receive buffer is full increase its size
411  if (_recive_buffer.size() == size)
412  _recive_buffer.resize(2 * size);
413 
414  // Call the buffer received handler

416  }

420  TryReceive();
421  else
422  {

424  Disconnect(true);
425  }
426  });

433  void TrySend()
434  {
435  if (_sending)
436  return;
437 
438  if (!IsHandshaked())
439  return;
440 
441  // Swap send buffers
442  if (_send_buffer_flush.empty())
443  {
444  std::lock_guard<std::mutex> locker(_send_lock);
445 
446  // Swap flush and main buffers
447  _send_buffer_flush.swap(_send_buffer_main);
448  _send_buffer_flush_offset = 0;
449 
450  // Update statistic
451  _bytes_pending = 0;
452  _bytes_sending += _send_buffer_flush.size();
453  }
454  else
455  return;
456 
457  // Check if the flush buffer is empty
458  if (_send_buffer_flush.empty())
459  {
460  // Call the empty send buffer handler
461  onEmpty();
462  return;
463  }
464 
465  // Async write with the write handler
466  _sending = true;
467  auto self(this->shared_from_this());
468  auto async_write_handler = make_alloc_handler(_send_storage, [this, self](std::error_code ec, size_t size)
469  {
470  _sending = false;
471 
472  if (!IsConnected())
473  return;
474 
475  // Send some data to the client
476  if (size > 0)
477  {
478  // Update statistic
479  _bytes_sending -= size;
480  _bytes_sent += size;
481 
482  // Increase the flush buffer offset
483  _send_buffer_flush_offset += size;
484 
485  // Successfully send the whole flush buffer
486  if (_send_buffer_flush_offset == _send_buffer_flush.size())
487  {
488  // Clear the flush buffer
489  _send_buffer_flush.clear();
490  _send_buffer_flush_offset = 0;
491  }
492 
493  // Call the buffer sent handler
494  onSent(size, bytes_pending());
495  }
496 
497  // Try to send again if the session is valid
498  if (!ec)
499  TrySend();
500  else
501  {
502  SendError(ec);
503  Disconnect(true);
504  }
505  });
506  if (_strand_required)
507  asio::async_write(_stream, asio::buffer(_send_buffer_flush.data() + _send_buffer_flush_offset, _send_buffer_flush.size() - _send_buffer_flush_offset), bind_executor(_strand, async_write_handler));
508  else
509  asio::async_write(_stream, asio::buffer(_send_buffer_flush.data() + _send_buffer_flush_offset, _send_buffer_flush.size() - _send_buffer_flush_offset), async_write_handler);
510  }

528  void SendError(std::error_code ec)
529  {
530  // Skip Asio disconnect errors
531  if ((ec == asio::error::connection_aborted) ||
532  (ec == asio::error::connection_refused) ||
533  (ec == asio::error::connection_reset) ||
534  (ec == asio::error::eof) ||
535  (ec == asio::error::operation_aborted))
536  return;
537 
538  // Skip OpenSSL annoying errors
539  if (ec == asio::ssl::error::stream_truncated)
540  return;
541  if (ec.category() == asio::error::get_ssl_category())
542  {
543  if ((ERR_GET_REASON(ec.value()) == SSL_R_DECRYPTION_FAILED_OR_BAD_RECORD_MAC) ||
544  (ERR_GET_REASON(ec.value()) == SSL_R_PROTOCOL_IS_SHUTDOWN) ||
545  (ERR_GET_REASON(ec.value()) == SSL_R_WRONG_VERSION_NUMBER))
546  return;
547  }
548 
549  onError(ec.value(), ec.category().name(), ec.message());
550  }
