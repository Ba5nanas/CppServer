88 void WSClient::onReceivedResponseHeader(const HTTP::HTTPResponse& response)
89 {
90  // Check for WebSocket handshaked status
91  if (_handshaked)
92  return;
93 
94  // Try to perform WebSocket handshake
95  if (response.status() == 101)
96  {
97  bool error = false;
98  bool accept = false;
99  bool connection = false;
100  bool upgrade = false;
101 
102  for (size_t i = 0; i < response.headers(); ++i)
103  {
104  auto header = response.header(i);
105  auto key = std::get<0>(header);
106  auto value = std::get<1>(header);
107 
108  if (key == "Connection")
109  {
110  if (value != "Upgrade")
111  {
112  error = true;
113  onError(asio::error::fault, "WebSocket error", "Invalid WebSocket handshaked response: 'Connection' header value must be 'Upgrade'");
114  break;
115  }
116 
117  connection = true;
118  }
119  else if (key == "Upgrade")
120  {
121  if (value != "websocket")
122  {
123  error = true;
124  onError(asio::error::fault, "WebSocket error", "Invalid WebSocket handshaked response: 'Upgrade' header value must be 'websocket'");
125  break;
126  }
127 
128  upgrade = true;
129  }
130  else if (key == "Sec-WebSocket-Accept")
131  {
132  // Calculate the original WebSocket hash
133  std::string wskey = CppCommon::Encoding::Base64Encode(id().string()) + "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
134  char wshash[SHA_DIGEST_LENGTH];
135  SHA1((const unsigned char*)wskey.data(), wskey.size(), (unsigned char*)wshash);
136 
137  // Get the received WebSocket hash
138  wskey = CppCommon::Encoding::Base64Decode(value);
139 
140  // Compare original and received hashes
141  if (std::strncmp(wskey.data(), wshash, std::max(wskey.size(), sizeof(wshash))) != 0)
142  {
143  error = true;
144  onError(asio::error::fault, "WebSocket error", "Invalid WebSocket handshaked response: 'Sec-WebSocket-Accept' value validation failed");
145  break;
146  }
147 
148  accept = true;
149  }
150  }
151 
152  // Failed to perfrom WebSocket handshake
153  if (!accept || !connection || !upgrade)
154  {
155  if (!error)
156  onError(asio::error::fault, "WebSocket error", "Invalid WebSocket response");
157  DisconnectAsync();
158  return;
159  }
160 
161  // WebSocket successfully handshaked!
162  _handshaked = true;
163  _mask = rand();
164  onWSConnected(response);
165 
166  return;
167  }
168 
169  // Disconnect on WebSocket handshake
170  onError(asio::error::fault, "WebSocket error", "Invalid WebSocket response status: {}"_format(response.status()));
171  DisconnectAsync();
172 }
